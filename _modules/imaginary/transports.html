class MockTransport(Transport):

    def get(self, url: Text, params: Params) -> Response:
        logger.info('mocking GET %s (%s)', url, params)
        return Response()

    def post(
        self,
        url: Text,
        params: Params = None,
        data: Params = None,
        files: Params = None,
    ) -> Response:
        logger.info(
            'mocking POST %s (params: %s, data: %s)',
            url,
            params,
            data,
        )
        return Response()
